DROP TABLE IF EXISTS `Orders`;
CREATE TABLE `orders` (
	`Id` BIGINT(20) NOT NULL AUTO_INCREMENT,
	`AmzOrderId` VARCHAR(20) NOT NULL COLLATE 'utf8_unicode_ci',
	`AccountId` VARCHAR(20) NOT NULL COLLATE 'utf8_unicode_ci',
	`PurchasedDate` DATETIME NOT NULL,
	`Country` VARCHAR(20) NOT NULL COLLATE 'utf8_unicode_ci',
	`ShipAddress` VARCHAR(200) NOT NULL COLLATE 'utf8_unicode_ci',
	`ShipPhone` VARCHAR(15) NOT NULL COLLATE 'utf8_unicode_ci',
	`Status` VARCHAR(20) NOT NULL COLLATE 'utf8_unicode_ci',
	`AliId` VARCHAR(20) NULL DEFAULT NULL COLLATE 'utf8_unicode_ci',
	`AliDate` DATETIME NULL DEFAULT NULL,
	`AliCashAmount` DECIMAL(10,0) NULL DEFAULT NULL,
	`AliTrackingNumber` VARCHAR(20) NULL DEFAULT NULL COLLATE 'utf8_unicode_ci',
	`AliTrackingDate` DATETIME NULL DEFAULT NULL,
	`Refund` DECIMAL(10,0) NULL DEFAULT NULL,
	`Note` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8_unicode_ci',
	PRIMARY KEY (`Id`)
)
COLLATE='utf8_unicode_ci'
ENGINE=InnoDB
AUTO_INCREMENT=1
;

DROP TABLE IF EXISTS `Items`;
CREATE TABLE `Items` (
	`Id` BIGINT(20) UNSIGNED NOT NULL AUTO_INCREMENT,
	`OrderId` BIGINT(20) UNSIGNED NOT NULL,
	`Image` VARCHAR(100) NULL DEFAULT NULL,
	`Asin` VARCHAR(50) NOT NULL,
	`Sku` VARCHAR(50) NOT NULL,
	`Quantity` TINYINT UNSIGNED NOT NULL,
	`UnitPrice` DECIMAL(20,5) UNSIGNED NULL DEFAULT NULL,
	`SubTotal` DECIMAL(20,5) UNSIGNED NULL DEFAULT NULL,
	PRIMARY KEY (`Id`)
)
COLLATE='utf8_unicode_ci'
ENGINE=InnoDB
AUTO_INCREMENT=1
;
